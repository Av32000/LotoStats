                    initResultFullVideo();

                    function initResultFullVideo() {
                        if(document.querySelectorAll('.result-full__video-btn').length > 0) {
                            let drawingPlayers = handleYoutubeApi();

                            addMultipleEvent(document.querySelectorAll('.result-full__video-btn'), 'click', function() {
                                handleVideoContainer(drawingPlayers, this);
                            });
                        }
                    }

                    function handleYoutubeApi() {
                        // This code loads the IFrame Player API code asynchronously. This is the Youtube-recommended script loading method
                        let tagForYtApi = document.createElement('script'),
                            firstScriptTag = document.getElementsByTagName('script')[0];

                        tagForYtApi.src = 'https://www.youtube.com/iframe_api';
                        tagForYtApi.id = 'youtubeScript';
                        firstScriptTag.parentNode.insertBefore(tagForYtApi, firstScriptTag);

                        // Get all drawing iframes
                        let drawingPlayers = [],
                            drawingPlayer,
                            drawingIframeIds = [],
                            drawingIframes = document.querySelectorAll('.result-full__video-container iframe');

                        Array.prototype.forEach.call(drawingIframes, function(drawingIframe) {
                            drawingIframeIds.push(drawingIframe.id);
                        });

                        // Create youtube player (function called by YouTube API)
                        window.onYouTubeIframeAPIReady = function() {
                            Array.prototype.forEach.call(drawingIframeIds, function(drawingIframeId) {
                                drawingPlayer = new YT.Player(drawingIframeId, {
                                    events: {
                                        onReady: function() {
                                            onPlayerReady(drawingIframeId);
                                        }
                                    }
                                });

                                drawingPlayers.push(drawingPlayer); //in case of multiple videos
                            });
                        };

                        return drawingPlayers;
                    }

                    function handleVideoContainer(drawingPlayers, btn) {
                        let videoBtnActived = document.querySelector('.result-full__video-btn.is-active'),
                            playerIdClicked = btn.getAttribute('data-id');

                        if(videoBtnActived && videoBtnActived.getAttribute('data-id') !== playerIdClicked) {
                            videoBtnActived.classList.toggle('is-active');
                            document.querySelector('.result-full__video-container.is-open').classList.toggle('is-open');

                            drawingPlayers.find(function(drawingPlayer) {
                                return getPlayerPlaying(drawingPlayer, videoBtnActived.getAttribute('data-id'));
                            }).stopVideo(); //stopVideo function is from Youtube Api
                        }

                        if (btn.classList.contains('is-active')) {
                            drawingPlayers.find(function(drawingPlayer) {
                                return getPlayerPlaying(drawingPlayer, playerIdClicked);
                            }).stopVideo(); //stopVideo function is from Youtube Api
                        }

                        btn.classList.toggle('is-active');
                        document.querySelector('.result-full__video-container[data-id="' + playerIdClicked + '"]').classList.toggle('is-open');

                        function getPlayerPlaying(el, idToCompare) {
                            return el.playerInfo.videoData.video_id === idToCompare;
                        }
                    }

                    function onPlayerReady(drawingIframeId) {
                        let drawingVideoBtn = document.querySelector('.result-full__video-btn[data-id="' + drawingIframeId.split('drawing-')[1] + '"]');

                        drawingVideoBtn.removeChild(drawingVideoBtn.querySelector('.spinner-loader_wrapper'));
                        drawingVideoBtn.classList.remove('has-loader');
                    }

                    function addMultipleEvent(element, eventName, callback) {
                        Array.prototype.forEach.call(element, function(el) {
                            el.addEventListener(eventName, callback);
                        });
                    }
